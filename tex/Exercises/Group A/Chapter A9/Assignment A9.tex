\begin{problem}
    The equation of the plane $\Pi_1$ is $y + z = 0$ and the equation of the line $l$ is \[\frac{x-5}2 = \frac{y-2}{-1} = \frac{z-2}3.\] Find
    \begin{enumerate}
        \item the position vector of the point of intersection of $l$ and $\Pi_1$,
        \item the length of the perpendicular from the origin to $l$,
        \item the Cartesian equation for the plane $\Pi_2$ which contains $l$ and the origin,
        \item the acute angle between the planes $\Pi_1$ and $\Pi_2$, giving your answer correct to the nearest $0.1 \deg$.
    \end{enumerate}
\end{problem}
\begin{solution}
    Note that $\Pi_1$ and $l$ have equations \[\vec r \dotp \cveciii011 = 0 \quad \tand \quad \vec r = \cveciii522 + \l \cveciii2{-1}3\] respectively, where $\l \in \RR$.

    \begin{ppart}
        Let $P$ be the point of intersection of $\Pi_1$ and $l$. Then we may write \[\oa{OP} = \cveciii522 + \l \cveciii2{-1}3\] for some $\l \in \RR$. Also, $\oa{OP} \dotp \cveciiix011 = 0$. Hence, \[\bs{\cveciii522 + \l \cveciii2{-1}3} \dotp \cveciii011 = 0.\] This gives $4 + 2\l = 0$, whence $\l = -2$. Thus, \[\oa{OP} = \cveciii522 - 2\cveciii2{-1}3 = \cveciii14{-4}.\]
    \end{ppart}
    \begin{ppart}
        The perpendicular distance from the origin to $l$ is \[\frac{\abs{\cveciiix522 \crossp \cveciiix2{-1}3}}{\abs{\cveciiix2{-1}3}} = \frac1{\sqrt{14}} \abs{\cveciii8{-11}{-9}} = \frac{\sqrt{266}}{\sqrt{14}} = \sqrt{19} \units.\]
    \end{ppart}
    \begin{ppart}
        Observe that $\Pi_2$ is parallel to $\cveciiix522$ and $\cveciiix2{-1}3$. Thus, \[\vec n = \cveciii522 \crossp \cveciii2{-1}3 = \cveciii8{-11}{-9}.\] Since $\Pi_2$ contains the origin, $d = 0$, it has vector equation \[\vec r \dotp \cveciii8{-11}{-9} = 0,\] which translates to $8x - 11y - 9z = 0$.
    \end{ppart}
    \begin{ppart}
        Let the acute angle between the planes be $\t$. Then \[\cos\t = \frac{\abs{\cveciiix011 \dotp \cveciiix8{-11}{-9}}}{\abs{\cveciiix011}\abs{\cveciiix8{-11}{-9}}} = \frac{20}{\sqrt{2}\sqrt{266}},\] so $\t = 29.9\deg \todp{1}$.
    \end{ppart}
\end{solution}

\begin{problem}
    The plane $\Pi_1$ has equation $\vec r \dotp (-\vec i + 2 \vec k) = -4$ and the points $A$ and $P$ have position vectors $4\vec i$ and $\vec i + \a \vec j + \vec k$ respectively, where $\a \in \RR$.

    \begin{enumerate}
        \item Show that $A$ lies on $\Pi_1$, but $P$ does not.
        \item Find, in terms of $\a$, the position vector of $N$, the foot of perpendicular of $P$ on $\Pi_1$.
    \end{enumerate}

    The plane $\Pi_2$ contains the points $A$, $P$ and $N$.
    \begin{enumerate}
        \setcounter{enumi}{2}
        \item Show that the equation of $\Pi_2$ is $\vec r \dotp (2\a \vec i + 5 \vec j + \a \vec k) = 8\a$ and write down the equation of $l$, the line of the intersection of $\Pi_1$ and $\Pi_2$.
    \end{enumerate}

    The plane $\Pi_3$ has equation $\vec r \dotp (\vec i + \vec j + 2\vec k) = 4$.
    \begin{enumerate}
        \setcounter{enumi}{3}
        \item By considering $l$, or otherwise, find the value of $\a$ for which the three planes intersect in a line.
    \end{enumerate}
\end{problem}
\begin{solution}
    Note that $\Pi_1$ has equation \[\vec r \dotp \cveciii{-1}02 = -4,\] and that $\oa{OA} = \cveciiix400$ and $\oa{OP} = \cveciiix1\a1$.

    \begin{ppart}
        Since \[\oa{OA} \dotp \cveciii{-1}02 = \cveciii400 \dotp \cveciii{-1}02 = -4,\] it follows that $A$ lies on $\Pi_1$. On the other hand, since \[\oa{OP} \dotp \cveciii{-1}02 = \cveciii1\a1 \dotp \cveciii{-1}02 = 1 \neq -4,\] it follows that $P$ does not lie on $\Pi_1$.
    \end{ppart}
    \begin{ppart}
        Note that $\oa{NP} = \l \cveciiix{-1}02$ for some real $\l$, and $\oa{ON} \dotp \cveciiix{-1}02 = -4$. Hence, \[\oa{NP} = \cveciii1\a1 - \oa{ON} = \l \cveciii{-1}02.\] Thus, \[\bs{\cveciii1\a1 - \oa{ON}} \dotp \cveciii{-1}02 = \l \cveciii{-1}02 \dotp \cveciii{-1}02.\] This gives $1 - (-4) = 5\l$, whence $\l = 1$. Hence, $\oa{NP} = \cveciiix{-1}02$, thus \[\oa{ON} = \oa{OP} - \oa{NP} = \cveciii2\a{-1}.\]
    \end{ppart}
    \begin{ppart}
        Note that $\Pi_2$ is parallel to $\oa{NP} = \cveciiix{-1}02$ and $\oa{AN} = \cveciiix{-2}\a{-1}$. Since \[\cveciii{-1}02 \crossp \cveciii{-2}\a{-1} = -\cveciii{2\a}5\a,\] we take $\vec n = \cveciiix{2\a}5\a$, whence \[d = \cveciii400 \dotp \vec n = 8\a.\] Thus, $\Pi_2$ has vector equation \[\vec r \dotp \cveciii{2\a}5{\a} = 8\a,\] which translates to $\vec r \dotp (2\a \vec i + 5 \vec j + \a \vec k) = 8\a$.

        Meanwhile, $l$, the line of intersection between $\Pi_1$ and $\Pi_2$ has equation \[\cveciii400 + \m \cveciii{-2}\a{-1}, \quad \m \in \RR.\]
    \end{ppart}
    \begin{ppart}
        If the three planes intersect in a line, they must intersect at $l$. Hence, $l$ lies on $\Pi_3$. This gives \[\bs{\cveciii400 + \m \cveciii{-2}\a{-1}} \dotp \cveciii112 = 4,\] so $4 + (\a - 4)\m = 4$, whence $(\a - 4)\m = 0$. Since this must hold for all $\m \in \RR$, we must have $\a = 4$.
    \end{ppart}
\end{solution}

\begin{problem}
    When a light ray passes from air to glass, it is deflected through an angle. The light ray $ABC$ starts at point $A(1, 2, 2)$ and enters a glass object at point $B(0, 0, 2)$. The surface of the glass object is a plane with normal vector $\vec n$. The diagram shows a cross-section of the glass object in the plane of the light ray and $\vec n$.

    \begin{center}\tikzsetnextfilename{149}

            \coordinate[label=right:$A$] (A) at (2, 1);
            \coordinate[label=above left:$B$] (B) at (0, 0);

            \coordinate (P1) at (3, 0);

            \draw (-3, 0) -- (P1);

            \draw pic [draw, angle radius=12mm, "$\t$"] {angle = P1--B--A};

    \end{center}

    \begin{enumerate}
        \item Find a vector equation of the line $AB$.
    \end{enumerate}

    The surface of the glass object is a plane with equation $x + z = 2$. $AB$ makes an acute angle $\t$ with the plane.

    \begin{enumerate}
        \setcounter{enumi}{1}
        \item Calculate the value of $\t$, giving your answer in degrees.
    \end{enumerate}

    The line $BC$ makes an angle of $45 \deg$ with the normal to the plane, and $BC$ is parallel to the unit vector $\cveciiix{-2/3}{p}{q}$.

    \begin{enumerate}
        \setcounter{enumi}{2}
        \item By considering a vector perpendicular to the plane containing the light ray and $\vec n$, or otherwise, find the values of $p$ and $q$.
    \end{enumerate}

    The light ray leaves the glass object through a plane with equation $3x + 3z = -4$.
    \begin{enumerate}
        \setcounter{enumi}{3}
        \item Find the exact thickness of the glass object, taking one unit as one cm.
        \item Find the exact coordinates of the point at which the light ray leaves the glass object.
    \end{enumerate}
\end{problem}
\begin{solution}
    Let $\Pi_G$ be the plane representing the surface of the glass object.

    \begin{ppart}
        Note that $\oa{AB} = -\cveciiix120$. Hence, the line $AB$ has vector equation \[\vec r = \cveciii002 + \l \cveciii120, \quad \l \in \RR.\]
    \end{ppart}
    \begin{ppart}
        Observe that $\Pi_G$ has equation \[\vec r \dotp \cveciii101 = 2.\] Hence, \[\sin \t = \frac{\abs{\cveciiix101 \dotp \cveciiix120}}{\abs{\cveciiix101} \abs{\cveciiix120}} = \frac1{\sqrt2 \sqrt5},\] so $\t = 71.6 \deg \todp{1}$.
    \end{ppart}
    \begin{ppart}
        Since line $BC$ makes an angle of $45 \deg$ with $\vec n_G$, we have \[\frac1{\sqrt2} = \sin 45\deg = \frac{\abs{\cveciiix101 \dotp \cveciiix{-2/3}{p}{q}}}{\abs{\cveciiix101} \abs{\cveciiix{-2/3}{p}{q}}} = \frac{\abs{q - 2/3}}{\sqrt2},\] so \[\abs{q - \frac23} = 1.\] Hence, $q = -1/3$. Note that we reject $q = 5/3$ since $\cveciiix{-2/3}{p}{q}$ is a unit vector, which implies that $\abs{q} \leq 1$.

        Let $\Pi_L$ be the plane containing the light ray. Note that $\Pi_L$ is parallel to $\oa{AB}$ and $\oa{BC}$. Hence, \[\vec n_L = \cveciii120 \crossp \cveciii{-2/3}{p}{q} = \frac13 \cveciii{6q}{-3q}{3p + 4}.\] Since $\Pi_L$ contains $\vec n_G$, it follows that $\vec n_L$ is perpendicular to $\vec n_G$, i.e. $\vec n_L \dotp \vec n_G = 0$. Thus, \[\cveciii{6q}{-3q}{3p + 4} \cdot \cveciii101 = 0.\] This gives $6q + 3p + 4 = 0$. Since $q = -1/3$, we have $p = -2/3$.
    \end{ppart}
    \begin{ppart}
        Let $\Pi_G'$ be the plane with equation $3x + 3z = -4$. Observe that $\Pi_G$ is parallel to $\Pi_G'$. Also note that $\bp{-4/3, 0, 0}$ is a point on $\Pi_G'$. Hence, the distance between $\Pi_G$ and $\Pi_G'$ is given by \[\frac{\abs{2 - \cveciiix{-4/3}00 \dotp \cveciiix101}}{\abs{\cveciiix101}} = \frac{10}{3\sqrt2} \text{ cm}.\]
    \end{ppart}
    \begin{ppart}
        Observe that \[\cveciii{-2/3}{p}{q} = \cveciii{-2/3}{-2/3}{-1/3} = -\frac13 \cveciii221,\] whence the line $BC$ has equation $\vec r  = \cveciiix002 + \m \cveciiix221$, where $\m \in \RR$. Let $P$ be the intersection between line $BC$ and $\Pi_G'$. Note that $\oa{OP} = \cveciiix002 + \m \cveciiix221$ for some $\m \in \RR$, and $\oa{OP} \dotp \cveciiix303 = -4$. Hence, \[\bs{\cveciii002 + \m \cveciii221} \dotp \cveciii303 = -4.\] This gives $6 - 9\m = -4$, so $\m = -10/9$. Hence, \[\oa{OP} = \cveciii002 - \frac{10}9 \cveciii221 = \cveciii{-20/9}{-20/9}{8/9}.\] The coordinates of the point are hence $(-20/9, -20/9, 8/9)$.
    \end{ppart}
\end{solution}
