\section{Assignment A8}

\begin{problem}
    Find the position vector of the foot of the perpendicular from the point with position vector $\vec c$ to the line with equation $\vec r = \vec a + \l \vec b$, $\l \in \RR$. Leave your answers in terms of $\vec a$, $\vec b$ and $\vec c$.
\end{problem}
\begin{solution}
    Let the foot of the perpendicular be $F$. We have that $\oa{OF} = \vec a + \l \vec b$ for some real $\l$, and $\oa{CF} \dotp \vec b = 0$. Note that $\oa{CF} = \oa{OF} - \oa{OC} = \vec a + \l \vec b - \vec c$. Thus, \[\oa{CF} \dotp \vec b = \bp{\vec a + \l \vec b - \vec c} \dotp \vec b = \l \abs{\vec b}^2 + (\vec a - \vec c) \dotp \vec b = 0,\] from which we obtain \[\l = \frac{(\vec c - \vec a) \dotp \vec b}{\abs{\vec b}^2}.\] Thus, \[\oa{OF} = \vec a + \bp{\frac{(\vec c - \vec a) \dotp \vec b}{\abs{\vec b}^2}} \vec b.\]
\end{solution}

\begin{problem}
    The point $O$ is the origin, and points $A$, $B$, $C$ have position vectors given by $\oa{OA} = 6\vec i$, $\oa{OB} = 3 \vec j$, $\oa{OC} = 4\vec k$. The point $P$ is on the line $AB$ between $A$ and $B$, and is such that $AP = 2PB$. The point $Q$ has position vector given by $\oa{OQ} = q\vec i$, where $q$ is a scalar.

    \begin{enumerate}
        \item Express, in terms of $\vec i$, $\vec j$, $\vec k$, the vector $\oa{CP}$.
        \item Show that the line $BQ$ has equation $\vec r = 3\vec j + t(q \vec i - 3\vec j)$, where $t$ is a parameter. Give an equation of the line $CP$ in a similar form.
        \item Find the value of $q$ for which the lines $CP$ and $BQ$ are perpendicular.
        \item Find the sine of the acute angle between the lines $CP$ and $BQ$ in terms of $q$.
    \end{enumerate}
\end{problem}
\begin{solution}
    We have that $\oa{OA} = \cveciiix600$, $\oa{OB} = \cveciiix030$ and $\oa{OC} = \cveciiix004$.

    \begin{ppart}
        By the ratio theorem, \[\oa{OP} = \frac{2\,\oa{OB} + \oa{OA}}{1 + 2} = \frac13 \bs{2\cveciii030 + \cveciii600} = \cveciii220,\] so $\oa{CP} = \cveciiix22{-4} = 2\vec i + 2\vec j - 4\vec k$.
    \end{ppart}
    \begin{ppart}
        Note that $\oa{BQ} = \oa{OQ} - \oa{OB} = \cveciiix{q}{-3}0$. Thus, $BQ$ is given by \[ \vec r = \cveciii030 + t\cveciii{q}{-3}0, \quad t \in \RR.\]
        
        Note that $\oa{CP} = \cveciiix22{-4} = 2\cveciiix11{-2}$. Hence, $CP$ is given by \[\vec r = \cveciii004 + u\cveciii11{-2}, \quad u \in \RR.\]
    \end{ppart}
    \begin{ppart}
        Since $CP$ is perpendicular to $BQ$, we have $\oa{CP} \dotp \oa{BQ} = 0$. Thus, \[2\cveciii11{-2} \dotp \cveciii{q}{-3}0 = q - 3 + 0 = 0,\] so $q = 3$.
    \end{ppart}
    \begin{ppart}
        Let $\t$ be the acute angle between $CP$ and $BQ$. Then \[\sin\t = \frac{\abs{\cveciiix11{-2} \crossp \cveciiix{q}{-3}0}}{\abs{\cveciiix11{-2}} \abs{\cveciiix{q}{-3}0}} = \frac{\abs{\cveciiix{-6}{2q}{3-q}}}{\sqrt6 \sqrt{q^2 + 9}} = \sqrt{\frac{5q^2 - 6q + 45}{6q^2 + 54}}.\]
    \end{ppart}
\end{solution}

\begin{problem}
    Line $l_1$ passes through the point $A$ with position vector $3\vec i - 2\vec k$ and is parallel to $-2\vec i + 4\vec j - \vec j$. Line $l_2$ has Cartesian equation given by $(x-1)/2 = y = z + 3$.

    \begin{enumerate}
        \item Show that the two lines intersect and find the coordinates of their point of intersection.
        \item Find the acute angle between the two lines $l_1$ and $l_2$. Hence, or otherwise, find the shortest distance from point $A$ to line $l_2$.
        \item Find the position vector of the foot $N$ of the perpendicular from $A$ to the line $l_2$. The point $B$ lies on the line $AN$ produced and is such that $N$ is the mid-point of $AB$. Find the position vector of $B$.
    \end{enumerate}
\end{problem}
\begin{solution}
    We have \[l_1 : \vec r = \cveciii30{-2} + \l\cveciii{-2}4{-1} \quad \tand \quad l_2 : \vec r = \cveciii10{-3} + \m\cveciii211,\] where $\l, \m \in \RR$.

    \begin{ppart}
        Consider the intersection of $l_1$ and $l_2$: \[\cveciii30{-2} + \l\cveciii{-2}4{-1} = \cveciii10{-3} + \m\cveciii211.\] This gives the system \[\systeme[\l\m]{2\l+2\m=2,{-4\l}+{\m}=0,{\l}+{\m}=1}\] which has the unique solution $\m = 4/5$ and $\l = 1/5$. Thus, the intersection point $P$ has position vector \[\cveciii30{-2} + \frac15\cveciii{-2}4{-1} = \frac15\cveciii{13}{4}{-11}\] and thus has coordinates $\bp{13/5, 4/5, -11/5}$.
    \end{ppart}
    \begin{ppart}
        Let $\t$ be the acute angle between $l_1$ and $l_2$. Then \[\cos\t = \frac{\abs{\cveciiix{-2}4{-1} \dotp \cveciiix211}}{\abs{\cveciiix{-2}4{-1}} \abs{\cveciiix211}} = \frac1{\sqrt{126}},\] so $\t = 84.9\deg \todp{1}$.

        Note that \[AP = \sqrt{\bp{\frac{17}5 - 3}^2 + \bp{-\frac45 - 0}^2 + \bp{-\frac95 - (-2)}^2} = \sqrt{\frac{21}{25}} = \frac{\sqrt{21}}5.\]

        Since $\sin\t = AN/AP$, we have that $AN = AP\sin\t$. Note that \[\sin\t = \sin \arccos \frac1{\sqrt{126}} = \frac{\sqrt{\bp{\sqrt{126}}^2 - 1}}{\sqrt{126}} = \frac{\sqrt{125}}{\sqrt{126}} = \frac{5\sqrt5}{\sqrt6 \sqrt{21}}.\] Thus, \[AN = \frac{\sqrt{21}}5 \cdot \frac{5\sqrt5}{\sqrt6 \sqrt{21}} = \sqrt{\frac56}.\] The shortest distance between $A$ and $l_2$ is hence $\sqrt{5/6}$ units.
    \end{ppart}
    \begin{ppart}
        Since $N$ is on $l_2$, we have that \[\oa{ON}= \cveciii10{-3} + \m\cveciii211\] for some real $\m$. Additionally, since $\oa{AN} \perp l_2$, we have $\oa{AN} \dotp \cveciiix211 = 0$. Since \[\oa{AN} = \bs{\cveciii10{-3} + \m\cveciii211} - \cveciii30{-2} = \cveciii{-2+2\m}{\m}{-1+\m},\] we have  \[\cveciii{-2+2\m}{\m}{-1+\m} \dotp \cveciii211 = -5 + 6\m = 0,\] so $\m = 5/6$. Hence, \[\oa{ON} = \cveciii10{-3} + \frac56 \cveciii211 = \frac16 \cveciii{16}5{-13}.\]

        Note that $\oa{ON} = (\oa{OA} + \oa{OB})/2$. Hence, \[\oa{OB} = 2\oa{ON} - \oa{OA} = \frac26\cveciii{16}5{-13} - \cveciii30{-2} = \frac13\cveciii75{-7}.\]
    \end{ppart}
\end{solution}
