\begin{problem}
    Expand $(1 + 2x)^{-1/3}$, where $\abs{x} < 1/2$, as a series of ascending powers of $x$, up to an including the term in $x^2$, simplifying the coefficients.

    By choosing $x = 1/14$, find an approximate value of $\sqrt[3]{7}$ in the form $p/q$, where $p$ and $q$ are to be determined.

    Using your calculator, calculate the numerical value of $\sqrt[3]{7}$. Compare this value to the approximate value found, and with reference to the value of $x$ chosen, comment on the accuracy of your approximation.
\end{problem}
\begin{solution}
    \[(1 + 2x)^{-1/3} = 1 + \frac{-1/3}{1!} (2x) + \frac{\bp{-1/3}\bp{-1/3 - 1}}2 (2x)^2 + \cdots = 1 - \frac23 x + \frac89 x^2 + \cdots.\] Substituting $x = 1/14$, \[\frac{\sqrt[3]{7}}{2} = \bs{1 + 2\bp{\frac1{14}}}^{-1/3} \approx 1 - \frac23 \bp{\frac1{14}} + \frac89 \bp{\frac1{14}}^2 = \frac{422}{441}.\] Thus, $\sqrt[3]{7} \approx 844/441 = 1.9138 \tosf{5}$. Since $\sqrt[3]7 = 1.9129 \tosf{5}$, the approximation is accurate.
\end{solution}

\begin{problem}
    In the triangle $ABC$, $AB = 1$, $BC = 3$ and angle $ABC = \t$ radians. Given that $\t$ is a sufficiently small angle, show that $AC \approx (4 + 3\t^2)^{\frac12} \approx a + b\t^2$ for constants $a$ and $b$ to be determined.
\end{problem}
\begin{solution}
    By the cosine rule, \[AC^2 = AB^2 + BC^2 - 2(AB)(BC)\cos{\angle ABC} = 1^2 + 3^2 - 2(1)(3) \cos \t = 10 - 6 \cos \t.\] Since $\t$ is sufficiently small, $\cos \t \approx 1 - \t^2/2$. Hence, \[AC^2 \approx 10 - 6\bp{1 - \frac{\t^2}{2}} = 4 + 3\t^2.\] We thus have \[AC = \bp{4 + 3\t^2}^{1/2} = 2\bp{1 + \frac{3\t^2}4}^{1/2} \approx 2\bs{1 + \frac12 \bp{\frac{3\t^2}4}} = 2 + \frac{3\t^2}4,\] so $a = 2$ and $b = 3/4$.
\end{solution}

\begin{problem}
    Given that $y = \ln \sec x$, show that

    \begin{enumerate}
        \item $\displaystyle \der[3]{y}{x} = 2 \der[2]{y}{x} \der{y}{x}$;
        \item the value of $\derx[4]{y}{x}$ when $x = 0$ is 2.
    \end{enumerate}

    Write down the Maclaurin series for $\ln \sec x$ up to and including the term in $x^4$. By substituting $x = \pi/4$, show that $\ln 2 \approx \pi^2/16 + \pi^4/1536$.
\end{problem}
\begin{solution}
    \begin{ppart}
        Differentiating, we see that $y' = \tan x$. Hence, $y'' = \sec^2 x = 1 + (y')^2$. Differentiating once more, we have $y''' = 2 y' y''$ as desired.
    \end{ppart}
    \begin{ppart}
        Implicitly differentiating the differential equation in part (a), we have $y^{(4)} = 2\bs{y''' \cdot y' + (y'')^2}$. At $x = 0$, we see that $y(0) = 0$, $y'(0) = 0$, $y''(0) = 1$, $y^{(3)}(0) = 0$, and $y^{(4)}(0) = 2$.
    \end{ppart}

    We have \[\ln \sec x = \frac{x^2}2  + \frac{x^4}{12} + \cdots.\] Substituting $x = \pi/4$, \[\frac12 \ln2 = \ln \sec \frac\pi4  \approx \frac12 \bp{\frac\pi4}^2 + \frac1{12} \bp{\frac\pi4}^4 = \frac{\pi^2}{32} + \frac{\pi^4}{3072},\] so $\ln 2 \approx \pi^2/16 + \pi^4/1536$.
\end{solution}
