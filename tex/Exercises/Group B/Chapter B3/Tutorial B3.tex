\begin{problem}
    The function $f$ is defined by \[f: x \mapsto x + \frac1{x}, \quad x \neq 0.\]

    \begin{enumerate}
        \item Sketch the graph of $f$ and explain why $\inv f$ does not exist.
        \item The function $h$ is defined by $h: x \mapsto f(x)$ with $x \in \RR$, $x \geq \a$, where $\a \in \RR^+$. Find the smallest value of $\a$ such that the inverse function of $h$ exists.
    \end{enumerate}

    Using this value of $\a$,
    
    \begin{enumerate}
        \setcounter{enumi}{2}
        \item State the range of $h$.
        \item Express $\inv h$ in a similar form and sketch on a single diagram, the graphs of $h$ and $\inv h$, showing clearly their geometrical relationship.
    \end{enumerate}
\end{problem}
\begin{solution}
    \begin{ppart}
        \begin{center}\tikzsetnextfilename{267}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    domain = -4:4,
                    samples = 81,
                    axis y line=middle,
                    axis x line=middle,
                    ytick = \empty,
                    xtick = \empty,
                    xlabel = {$x$},
                    ylabel = {$y$},
                    legend cell align={left},
                    legend pos=outer north east,
                    after end axis/.code={
                        \path (axis cs:0,0) 
                            node [anchor=south east] {$O$};
                        }
                    ]
                    \addplot[plotRed, unbounded coords=jump] {x + 1/x};
        
                    \addlegendentry{$y = f(x)$};

                    \addplot[dotted, thick] {x};

                    \node[anchor=south west, rotate=16] at (-4, -4) {$y=x$};

                    \fill (-1, -2) circle[radius=2.5 pt];
                    
                    \node[anchor=north, fill=white, opacity = 0.6, text opacity=1] at (-1.2, -2.2) {$(-1, -2)$};

                    \fill (1, 2) circle[radius=2.5 pt] node[anchor=south] {$(1, 2)$};
                \end{axis}
            \end{tikzpicture}
        \end{center}

        $y=f(x)$ does not pass the horizontal line test. Hence, $f$ is not one-one, so $\inv f$ does not exist.
    \end{ppart}
    \begin{ppart}
        We have $f'(x) = 1 - 1/x^2$. Consider $f'(x) = 0$ for $x > 0$. Then $x^2 = 1$, so $x = 1$. Looking at the graph of $y=f(x)$, we see that $f(x)$ achieves a minimum at $x = 1$. Hence, $f$ is increasing for all $x \geq 1$. Thus, the smallest value of $\a$ is 1.
    \end{ppart}
    \begin{ppart}
        Note $f(1) = 2$. Hence, from the graph, $\ran h = [2, \infty)$.
    \end{ppart}
    \begin{ppart}
        Let $y = h(x) = x + 1/x$. Then $xy = x^2 + 1$. By the quadratic formula, \[x = \frac{y \pm \sqrt{y^2-4}}{2}.\] Note that $f(2) = 5/2$, so we take the positive branch. Since $\dom{\inv f} = \ran f = [2, \infty)$, we thus have \[\inv h: x \mapsto \frac12\bp{x + \sqrt{x^2-4} }, \quad x \geq 2.\]
    \end{ppart}
\end{solution}

\begin{problem}
    The function $f$ is defined as $f: x \mapsto x^3 + x - 7$ for $x \in \RR$.

    \begin{enumerate}
        \item By using a graphical method or otherwise, show that the inverse of $f$ exists.
        \item Solve exactly the equation $\inv f(x) = 0$. Sketch the graph of $\inv f$ together with the graph of $f$ on the same diagram.
        \item Find, in exact form, the coordinates of the intersection point(s) of the graphs of $f$ and $\inv f$.
        \item Given that the gradient of the tangent to the curve with equation $y = \inv f(x)$ is $1/4$ at the point with $x = p$, find the possible values of $p$.
    \end{enumerate}
\end{problem}
\begin{solution}
    \begin{ppart}
        \begin{center}\tikzsetnextfilename{268}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    domain = -2.5:2.5,
                    samples = 161,
                    axis y line=middle,
                    axis x line=middle,
                    ytick = {-7},
                    xtick = \empty,
                    xlabel = {$x$},
                    ylabel = {$y$},
                    legend cell align={left},
                    legend pos=outer north east,
                    after end axis/.code={
                        \path (axis cs:0,0) 
                            node [anchor=north east] {$O$};
                        }
                    ]
                    \addplot[plotRed] {x^3 + x - 7};

                    \addlegendentry{$y = f(x)$};
                \end{axis}
            \end{tikzpicture}
        \end{center}

        $y = f(x)$ passes the horizontal line test. Hence, $f$ is one-one. Thus, $\inv f$ exists.
    \end{ppart}
    \begin{ppart}
        We have $\inv f(x) = 0$, so $x = f(0) = -7$.

        \begin{center}\tikzsetnextfilename{269}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    domain = -2.5:2.5,
                    samples = 161,
                    axis y line=middle,
                    axis x line=middle,
                    ytick = {-7},
                    xtick = {-7},
                    xlabel = {$x$},
                    ylabel = {$y$},
                    ymin=-10,
                    ymax=8,
                    xmin=-10,
                    xmax=8,
                    legend cell align={left},
                    legend pos=outer north east,
                    after end axis/.code={
                        \path (axis cs:0,0) 
                            node [anchor=south east] {$O$};
                        }
                    ]
                    \addplot[plotRed] {x^3 + x - 7};

                    \addlegendentry{$y = f(x)$};

                    \plot[plotBlue, samples=161] (\x^3 + \x -7, \x);

                    \addlegendentry{$y = \inv f(x)$};

                    \addplot[dotted, thick, domain=-10:10] {x};

                    \node[rotate=45, anchor=south] at (-7, -7) {$y=x$};
                \end{axis}
            \end{tikzpicture}
        \end{center}
    \end{ppart}
    \begin{ppart}
        Let ($\a$, $\b$) be the coordinates of the intersection between $f(x)$ and $\inv f(x)$. From the graph, we see that $\a=\b$, hence $f(\a) = \a$. Hence, $f(\a) = \a^3 + \a - 7 = \a$, which gives $\a^3 = 7$, whence $\a = \sqrt[3]{7}$. The coordinates are thus $\bp{\sqrt[3]{7}, \sqrt[3]{7}}$.
    \end{ppart}
    \begin{ppart}
        Note that \[\der{}{x} \inv f(x) = \frac1{f'(\inv f(x))}.\] Evaluating at $x = p$, we obtain \[\frac14 = \frac1{f'(\inv f(p))},\] so $f'(\inv f(p)) = 4$. Since $f'(x) = 3x^2 + 1$, we have $3\inv f(p)^2 + 1 = 4$, so $\inv f(p) = 1$ or $-1$. If $\inv f(p) = 1$, we have $p = f(1) = -5$. If $\inv f(p) = -1$, we have $p = f(-1) = -9$.
    \end{ppart}
\end{solution}

\begin{problem}
    The functions $f$ and $g$ are defined as follows:
    \begin{alignat*}{2}
        f &: x \mapsto 3 - x, &&\quad x \in \RR\\
        g &: x \mapsto \frac4{x}, &&\quad x \in \RR, \; x \neq 0
    \end{alignat*}

    \begin{enumerate}
        \item Show that the composite function $fg$ exists and express the definition of $fg$ in a similar form. Find the range of $fg$.
        \item Find, in similar form, $g^2$ and $g^3$, and deduce $g^{2017}$.
        \item Find the set of values of $x$ for which $g(x) = \inv g(x)$.
    \end{enumerate}
\end{problem}
\begin{solution}
    \begin{ppart}
        Note that $\ran g = \RR^{\ast}$ and $\dom g = \RR$. Hence, $\ran g \subseteq \dom g$. Thus, $fg$ exists. We see that $fg(x) = f(4/x) = 3 - 4/x$. Also, $\dom{fg} = \dom{g} = \RR^{\ast}$. Thus, \[fg: x \mapsto 3 - \frac4x, \quad x \in \RR^{\ast}.\]
        
        Since $4/x$ can take on any value except 0, then $fg(x) = 3 - 4/x$ can take on any value except 3. Thus, $\ran{fg} = \RR \setminus \bc{3}$.
    \end{ppart}
    \begin{ppart}
        We have $g^2(x) = g(4/x) = x$. Hence, $g^2: x \mapsto x$ for $x \in \RR^{\ast}$. We have $g^3(x) = g(g^2(x)) = g(x) = 4/x$. Hence, $g^3: x \mapsto 4/x$ for $x \in \RR^{\ast}$.

        It follows that \[g^n(x) = \begin{cases}
            x, & \text{$n$ even},\\
            4/x, & \text{$n$ odd}.
        \end{cases}\]

        Thus, $g^{2017}: x \mapsto 4/x$ for $x \in \RR^{\ast}$.
    \end{ppart}
    \begin{ppart}
        Note that $g(x) = \inv g(x)$ implies that $g^2(x) = x$, provided $g(x)$ is in $\dom g$. From the definition of $g^2(x)$, we know that $g^2(x) = x$ for all $x$ in $\dom{g^2} = \RR^{\ast}$. Since $g(x)$ is always non-zero, it follows that the solution set is $\RR^{\ast}$.
    \end{ppart}
\end{solution}

\begin{problem}
    \begin{enumerate}
        \item The function $f$ is given by $f: x \mapsto 1 + \sqrt{x}$ for $x \in \RR^+$. \begin{enumerate}
            \item Find $\inv f(x)$ and state the domain of $\inv f$.
            \item Find $f^2(x)$ and the range of $f^2$.
            \item Show that if $f^2(x) = x$ then $x^3 - 4x^2 + 4x - 1 = 0$. Hence, find the value of $x$ for which $f^2(x) = x$. Explain why this value of $x$ satisfies the equation $f(x) = \inv f(x)$.
        \end{enumerate}
        \item The function $g$, with domain the set of non-negative integers, is given by
        \[
            g(n) =
            \begin{cases}
                1, & n = 0\\
                2 + g\bp{n/2}, & n \text{ even}\\
                1 + g(n-1), & n \text{ odd}
            \end{cases}
        \]
        \begin{enumerate}
            \item Find $g(4)$, $g(7)$ and $g(12)$.
            \item Does $g$ have an inverse? Justify your answer.
        \end{enumerate}
    \end{enumerate}

\begin{solution}
