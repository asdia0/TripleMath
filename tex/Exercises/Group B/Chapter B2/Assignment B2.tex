\begin{problem}
    A graph with equation $y = f(x)$ undergoes transformation $A$ followed by transformation $B$ where $A$ and $B$ are described as follows:

    \begin{itemize}
        \item $A$: a translation of 1 unit in the positive direction of the $x$-axis
        \item $B$: a scaling parallel to the $x$-axis by a factor $1/2$
    \end{itemize}

    The resulting equation is $y = 4x^2-4x+1$. Find the equation $y = f(x)$.
\end{problem}
\begin{solution}
    We apply $B^{-1}$ followed by $A^{-1}$ to the curve given by $y = 4x^2 - 4x + 1$.

    \renewcommand{\theenumi}{\arabic{enumi}.}
    \begin{enumerate}
        \item[] \[y = 4x^2 - 4x + 1.\]
        \item Scale parallel to the $x$-axis by a factor of 2. \[y = x^2 - 2x + 1 = (x-1)^2. \tag{$x \mapsto x/2$}\]
        \item Translate 1 unit in the negative direction of the $x$-axis. \[y = x^2. \tag{$x \mapsto x+1$}\]
    \end{enumerate}
    \renewcommand{\theenumi}{(\alph{enumi})}

    Thus, $f(x) = x^2$.
\end{solution}
